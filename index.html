import requests from bs4 import BeautifulSoup url = "https://yallashootz.com/matches-yesterday/" response = requests.get(url) soup = BeautifulSoup(response.text, 'html.parser') matches = soup.find_all('div', class_='match-card') # تأكد من اسم الكلاس الحقيقي حسب الصفحة # HTML رأس الصفحة بتنسيق ListView html_output = """ مباريات الأمس - يلا شوت

مباريات الأمس

    """ # إضافة كل مباراة كعنصر داخل القائمة for match in matches: try: team1 = match.find('div', class_='team1').text.strip() team2 = match.find('div', class_='team2').text.strip() time = match.find('div', class_='match-time').text.strip() result = match.find('div', class_='result').text.strip() competition = match.find('div', class_='competition').text.strip() html_output += f"""
  • {team1} 🆚 {team2}
    النتيجة: {result}
    الوقت: {time}
    البطولة: {competition}
    • """ except: continue # نهاية الصفحة html_output += """

© تم الإنشاء بواسطة ChatGPT

""" # حفظ الملف النهائي with open("matches.html", "w", encoding="utf-8") as file: file.write(html_output) print("✅ تم إنشاء matches.html بنجاح بتنسيق ListView.")
