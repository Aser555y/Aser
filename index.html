from bs4 import BeautifulSoup # قراءة محتوى صفحة HTML with open("view-page-source.com-yallashootz.com_matches-yesterday_.html", "r", encoding="utf-8") as file: soup = BeautifulSoup(file, "html.parser") matches = [] # استخراج البيانات من عناصر المباريات for match in soup.select(".AY_Match.finished"): teams = match.select(".TM_Name") logos = match.select(".TM_Logo img") time = match.select_one(".MT_Time").text.strip() result = match.select_one(".MT_Result").text.strip() channel = match.select(".MT_Info li")[0].text.strip() commentator = match.select(".MT_Info li")[1].text.strip() tournament = match.select(".MT_Info li")[2].text.strip() matches.append({ "team1": teams[0].text.strip(), "team2": teams[1].text.strip(), "logo1": logos[0].get("data-src") or logos[0].get("src"), "logo2": logos[1].get("data-src") or logos[1].get("src"), "time": time, "result": result, "channel": channel, "commentator": commentator, "tournament": tournament, }) # إنشاء ملف HTML بالتنسيق المطلوب with open("matches.html", "w", encoding="utf-8") as f: f.write(""" مباريات الأمس
""") # إضافة كل مباراة for m in matches: f.write(f"""
{m['team1']}
{m['time']}
{m['result']}
{m['team2']}
🏆 {m['tournament']} | 📺 {m['channel']} | 🎤 {m['commentator']}
""") f.write("""
""") print("✅ تم إنشاء الملف matches.html بنجاح.")
